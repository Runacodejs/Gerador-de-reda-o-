async function gerarRedacao() { const tema = document.getElementById("tema").value; if (!tema) { alert("Digite um tema primeiro!"); return; } showLoader(); try { const res = await fetch("/api/redacao", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ tema }) }); const data = await res.json(); document.getElementById("saida").innerText = data.redacao || "Erro ao gerar redação."; } catch { document.getElementById("saida").innerText = "Erro ao gerar redação."; } } function copiarRedacao() { const texto = document.getElementById("saida").innerText; navigator.clipboard.writeText(texto).then(() => alert("Redação copiada!")); } function apagarRedacao() { document.getElementById("saida").innerText = "Sua redação aparecerá aqui..."; document.getElementById("tema").value = ""; }
